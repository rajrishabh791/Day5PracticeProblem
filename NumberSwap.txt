package com.bridgelabz;

public class NumberSwap {
    int a=6;
    int b=9;
    int c;
    void numberSwap()
    {
        System.out.println("Before Swapping " + " a " + a + " b " + b );
        c=a;
        a=b;
        b=c;
        System.out.println("After Swapping " + " a " + a + " b " + b );
    }
    public static void main(String args[])
    {
        NumberSwap n = new NumberSwap();
        n.numberSwap();
    }
}
