package com.bridgelabz;

public class QuotientRemainder {
    void quotientRemainder(int a)
    {
        int rem = a % 10;
        System.out.println("Remainder of " + a + " is " + rem);
        int quo = a / 10;
        System.out.println("Quotient of " + a + " is " + quo);
    }
    public static void main(String args[])
    {
        QuotientRemainder q = new QuotientRemainder();
        q.quotientRemainder(245);
        q.quotientRemainder(452);
        q.quotientRemainder(35);
        q.quotientRemainder(78);
    }
}
